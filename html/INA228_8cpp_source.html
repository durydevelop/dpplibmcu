23 /*
24  * SHUNT_CAL is a conversion constant that represents the shunt resistance
25  * used to calculate current value in Amps. This also sets the resolution
26  * (CURRENT_LSB) for the current register.
27  *
28  * SHUNT_CAL is 15 bits wide (0 - 32768)
29  *
30  * SHUNT_CAL = 13107.2 x 10^6 x CURRENT_LSB x Rshunt
31  *
32  * CURRENT_LSB = Max Expected Current / 2^19
33  */
34 
35 #define CURRENT_LSB 0.000015625
36 #define SHUNT_CAL 1024

58 float INA228::voltage(void)
59 {
60  int32_t iBusVoltage;
61  float fBusVoltage;
62  bool sign;
63 
64  askForBuf(devAddr,INA228_VBUS,(uint8_t *)&iBusVoltage, 3);
65  sign = iBusVoltage & 0x80;
66  iBusVoltage = bswap_32(iBusVoltage & 0xFFFFFF) >> 12;
67  if (sign) iBusVoltage += 0xFFF00000;
68  fBusVoltage = (iBusVoltage) * 0.0001953125;
69
