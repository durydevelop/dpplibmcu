286 /**
287  * @brief Reach Rpm speed using ramp (output directly on pin).
288  * Change speed from CurrRpm to Rpm in time of RampMillis.
289  * N.B.:
290  * -Ramp is calculated for both increase and decrese speed.
291  * -Motor rotation direction still remain to this RotationDir value.
292  * -One pin is used as emergency stop.
293  *
294  * e.g.
295  * @code {.C++}
296  // If Pin 2 wil trigger a LOW_LEVEL status funcion stops before Ramp time.
297  bool fault=Ramp(120,4000,CC,2,LOW_LEVEL); // Reach 120 rpm in 4 seconds
298  if (!fault) {
299  Ramp(60,2000,CC,2,LOW_LEVEL); // Decrease to 60 rpm in 2 seconds
300  }
301  if (!fault) {
302  Ramp(0,2000,CC,2,LOW_LEVEL); // Stop motor in 2 seconds
303  }
304  * @endcode
305  *
306  * @param RampMillis -> Duration of ramp in milliseconds. Setting a value if 0, immedialtly apply Rpm value.
307  * @param Rpm -> Rpm value to set.
308  * @param RotationDir -> Specify rotation direction ROT_CC or ROT_CW. ROT_CURRENT will not change it.
309  * @param FaultPin -> Pin used to emergency stop motor, when this pin is in FaultLevel, motor will be stopped.
310  * A value of -1 disable the check (!! BE CAREFULLY !!).
311  * @param FaultLevel -> The FaultPin level that triggers the emergency stop.
312  *
313  * @return false if FaultPin has triggered an emergency stop, otherwise, after RampMillis time, return true.
314  */
315 bool DStepper::RampRpm(uint16_t RampMillis, uint16_t Rpm, DRotationDir RotationDir, int8_t FaultPin, DPinLevel FaultLevel)
316 {
317  if (RampMillis > 0) {
318  unsigned long StartRamp=millis();
319  uint16_t CurrRpm=GetRpm();
320  int8_t Inc=1; // default increasing
321  if (Rpm < CurrRpm) {
322  // decreasing
323  Inc=-1;
324  }
325 
326  uint16_t DelayMillis=RampMillis/((Rpm-CurrRpm)*Inc); // Delay time after each rpm increment
327  if (RotationDir != ROT_CURRENT) {
328  SetDir(RotationDir);
329  }
330  dPwmOut->On();
331  while (millis()-StartRamp < RampMillis) {
332  CurrRpm+=Inc;
333  dPwmOut->SetFreq((CurrRpm*StepsPerRevol)/60,CurrDutyCycle);
334  unsigned long StartDelay=millis();
335  while (millis()-StartDelay < DelayMillis) {
336  if (FaultPin >= 0) {
337  if (ReadPin(FaultPin) == FaultLevel) {
338  dPwmOut->Off();
339  return false;
340  }
341  }
342  };
343  }
344  }
345  SetRpm(Rpm);
346  return true;
347 }
