126 /**
127  * @brief Set wheels balancing value for forward direction.
128  * A negative value means that sx wheel runs faster than dx, so, MotorDX::PwmLimitFw will decrease (but not under DMotor::MIN_CAL_PWM).
129  * A positive value means that dx wheel runs faster than sx, so, MotorSX::PwmLimitFw will decrease (but not under DMotor::MIN_CAL_PWM).
130  * A zero value means that sx and dx wheel runs without compensation.
131  *
132  * @param Value -> from -[DDCMotor::MAX_PWM-DDCMotor::MIN_CAL_PWM] to [DDCMotor::MAX_PWM-DDCMotor::MIN_CAL_PWM]. Default: -155 to 155
133  */
134 void DDCMotorWheels::SetBalancingFw(short int Value)
135 {
136  // Limit for calibration value
137  unsigned short Limit=DDCMotor::MAX_PWM-DDCMotor::MIN_CAL_PWM;
138  if (Value > Limit) Value=Limit;
139  else if (Value < -Limit) Value=-Limit;
140 
141  // Balancing=CalLimitSX-CalLimitDX
142  // so...
143  // CalLimitDX=CalLimitSX-Balancing
144  // CalLimitSX=CalLimitDX+Balancing
145  if (Value > 0) {
146  // To DX:
147  // SX no limit
148  // DX decrease limit
149  MotorSX->ResetCalLimitFw();
150  MotorDX->SetCalLimitFw(MotorSX->GetCalLimitFw()-Value);
151  }
152  else if (Value < 0) {
153  // To SX:
154  // SX decrese balancing
155  // DX no limit
156  MotorSX->SetCalLimitFw(MotorDX->GetCalLimitFw()+Value);
157  MotorDX->ResetCalLimitFw();
158  }
159  else {
160  MotorSX->ResetCalLimitFw();
161  MotorDX->ResetCalLimitFw();
162  }
163 }
