35 bool INA226::begin(void)
36 {
37  bool ret=writeWord(devAddr,INA226_CFG_REG, 0x8000); // Reset
38  ret&=writeWord(devAddr,INA226_CFG_REG, 0x4527); // Average over 16 Samples
39  ret&=writeWord(devAddr, INA226_CAL_REG, 1024); // 1A, 0.100Ohm Resistor
40  return ret;

56 float INA226::current()

60

62  // Internally Calculated as Current = ((ShuntVoltage * CalibrationRegister) / 2048)

69 float INA226::power()
70 {
71  int16_t iPower;
72  float fPower;
73 
74  iPower = askForWord(devAddr,INA226_POWER_REG);
75  // The Power Register LSB is internally programmed to equal 25 times the programmed value of the Current_LSB
