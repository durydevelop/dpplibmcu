18  DBL_PRESSED TRIGGER: press-release-press within DblPressSpeedMillis time.
19  DBL_PRESSING STATE: after DBL_PRESSED, if button still pressed, the state change in DBL_PRESSING until button is released.
20  RELEASED TRIGGER: release after PRESSED, LONG_PRESSED, DBL_PRESSED.

115 //! @return the current button State.
116 DDigitalButton::DButtonState DDigitalButton::read(void)
117 {
118  bool trig=false;
119 
120  if (prevState == RELEASED) {
121  currState=RELEASE;
122  }
123  if (prevState == PRESSED) {
124  currState=PRESS;
125  }
126 
127  lastResult=readPin(pin,handle);
128  if (lastResult < 0) {
129  return ERROR_STATE;
130  }
131 
132  if (lastResult == pressedLevel) {
133  pressMs=millis(); // Press time
134  if (prevState == RELEASE) {
135  if ((pressMs-releaseMs) > 50) {
136  currState=PRESS;
137  }
138  }
139  else if (prevState == PRESS) {
140  if ((pressMs-pressedMs) < dblPressSpeedDuration) {
141  currState=DBL_PRESSED;
142  trig=true;
143  }
144  if ((pressMs-releaseMs) > longPressedDuration) {
145  currState=LONG_PRESSED;
146  trig=true;
147  }
148  }
149  else if (prevState == LONG_PRESSED) {
150  currState=LONG_PRESSING;
151  }
152  else if (prevState == DBL_PRESSED) {
153  currState=DBL_PRESSING;
154  }
155  }
156  else {
157  releaseMs=millis(); // Release time
158  if (prevState == PRESS) {
159  if ((releaseMs-pressMs) > pressedDuration) {
160  currState=PRESSED;
161  trig=true;
162  pressMs=releaseMs;
163  }
164  else if ((releaseMs-pressMs) > 50) {
165  pressedMs=millis();
166  }
167  }
168  else if (prevState == PRESSED || prevState == LONG_PRESSING || prevState == LONG_PRESSED) {
169  currState=RELEASED;
170  trig=true;
171  }
172  else {
173  currState=RELEASE;
174  }
175  }
176 
177  prevState=currState;
178  if (trig) {
179  if (callback != NULL) {
180  callback(currState);
181  }
182  }
183 
184  return (currState);
185 }
